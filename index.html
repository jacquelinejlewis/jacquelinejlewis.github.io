C++ Snippet

Binary/Decimal validator (original assignment code, plus a tiny main() so it runs).

converter.cpp 
#include<iostream> //Jacqueline Lewis
#include <string>   //25010437
using namespace std;  //Manawatu campus

bool binaryQ (string x) {
    if (x.empty()) return false;
    if (x[0] != '0') {
        return false;
    }
    if (x.length() > 9) {
        return false;
    }
    for (int i = 0; i < (int)x.length(); i++) {
        if (x[i] != '0' && x[i] != '1') {
            return false; 
        }
    }
    return true;
}

bool decimalQ(string x) {
    if (x.empty()) return false;
    if (x.length() > 1 && x[0] == '0') {
        return false;
    }
    for (int i = 0; i < (int)x.length(); i++) {
        if (x[i] < '0' || x[i] > '9') {
            return false;
        }
    }
    try {
        int num = stoi(x);
        if (num < 0 || num > 255) {
            return false;
        }
    } catch (...) {
        return false;
    }
    return true;
}

int main() {
    string input;
    cout << "Enter a binary (starting with 0, up to 9 chars) or decimal (0..255): ";
    cin >> input;

    if (binaryQ(input)) {
        cout << input << " is a valid binary number." << endl;
    } else if (decimalQ(input)) {
        cout << input << " is a valid decimal number." << endl;
    } else {
        cout << "Invalid input." << endl;
    }
    return 0;
}
